package exp
import scala.quoted.*

trait MishaProvider {
  def misha: Int
}

object ApplyMaker {
  inline def forEnum[EN]: Any = ${ forEnumMacro[EN] }

  private def makeApply(using
      q: Quotes,
  )(
      clsSym: q.reflect.Symbol,
      caseSym: q.reflect.Symbol,
  ): List[q.reflect.Symbol] = {
    import q.reflect.*

    val applySym = Symbol.newMethod(
      clsSym,
      "misha",
      ByNameType(TypeRepr.of[Int]),
      Flags.Override,
      Symbol.noSymbol,
    )

    List(applySym)
  }

  private def forEnumMacro[EN: Type](using
      q: Quotes,
  ): Expr[Any] = {
    import q.reflect.*
    val enumSym = TypeRepr.of[EN].typeSymbol
    val cases = enumSym.children
    val mods = cases.flatMap { cs =>
      val objSym = Symbol.newModule(
        owner = Symbol.spliceOwner,
        name = cs.name,
        modFlags = Flags.EmptyFlags,
        clsFlags = Flags.EmptyFlags,
        parents = _ => List(TypeRepr.of[AnyRef], TypeRepr.of[MishaProvider]),
        decls = cls => makeApply(using q)(cls, cs),
        privateWithin = Symbol.noSymbol,
      )
      val methodSym =
        objSym.moduleClass.declarations.find(_.name == "misha").get
      val method = DefDef(methodSym, _ => Some('{ 42 }.asTerm))
      val clsDef = ClassDef(
        objSym.moduleClass,
        List(TypeTree.of[AnyRef], TypeTree.of[MishaProvider]),
        List(method),
      )
      val valDef = ValDef(objSym, None)
      List(clsDef, valDef)
    }
    val block = Block(mods, Literal(UnitConstant()))
    block.asExpr
  }
}
